\subsubsection*{Algorithmus}
\begin{algorithmic}
	\STATE  $newMark \leftarrow$ false
	\FORALL{pairs $(q, q')$ in $M$}
		\IF{$(q \in F \wedge q' \notin F) \vee (q \notin F \wedge q' \in F)$}
		\STATE $mark(q, q')$ \COMMENT{$M[q, q'] \leftarrow$ true}
		\STATE $newMark \leftarrow$ true
		\ENDIF
	\ENDFOR
	\WHILE{$newMark =$ true}
		\STATE $newMark \leftarrow$ false
		\FORALL{unmarked pairs $(q, q')$ in $M$}
			\IF{($\exists e \in \Sigma : (\delta(q, e), \delta(q', e))$ is marked)}
				\STATE $mark(q, q')$
				\STATE $newMark \leftarrow$ true
			\ENDIF
		\ENDFOR
	\ENDWHILE
\end{algorithmic}
